# Compute the Euclidean distance.

def compute_euclid(point, centroid):
    diff = (point - centroid)
    diff_sq = diff **2
    diff_sq_sum = np.sum(diff_sq)
    
    return diff_sq_sum

def mlp():
    '''
    2-layer Perceptron
    '''
    # Start
    model = Sequential()
    # Fully-Connected Layer -1
    model.add(Dense(30, input_dim=2, name='fc1'))
    # Activation
    model.add(Activation('relu'))
    # Fully-Connected Layer -2
    model.add(Dense(80, name='fc2'))
    # Activation
    model.add(Activation('relu'))
    # Dropout
    model.add(Dropout(0.5))
    # Fully-Connected Layer -4
    model.add(Dense(1, name='fc4'))
    
    return model
